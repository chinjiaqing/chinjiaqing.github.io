/**
 * 判断类型
 * @param {*} x 目标
 * @returns {string}
 */
function _typeof(x) {
	// 判断 null
	if (x === null) return "null";
	// 基本类型判断
	let t = typeof x;
	if (t !== "object") return t;
	// 判断对象类型
	t = Object.prototype.toString.call(x).slice(8, -1);
	// 判断包装类型
	if (["String", "Boolean", "Number"].includes(t)) return t;
	// 判断自定义构造函数创建的实例
	if (typeof x?.constructor?.name === "string") {
		return x.constructor.name;
	}
	return t.toLowerCase();
}
